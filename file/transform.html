用js控制css
var tX, tY, scX, scY, r, skX, skY;

var item = document.getElementById("item");

tX = document.getElementById("input_translateX").value;
tY = document.getElementById("input_translateY").value;
item.style.transform = "translate("+tX +"px,"+tY +"px)";

scX = document.getElementById("input_scaleX").value;
scY = document.getElementById("input_scaleY").value;
item.style.transform = "scale("+scX +","+scY +")";

r = document.getElementById("input_rotate").value;
item.style.transform = "rotate("+r+"deg)";

skX = document.getElementById("input_skewX").value;
skY = document.getElementById("input_skewY").value;
item.style.transform = "skew("+skX +"deg,"+skY +"deg)";
